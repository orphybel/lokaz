Voici la version modifiée de ton vite.config.jspour adapter ton site Bolt/Vite à GitHub Pages sur le chemin /lokaz/ :

js
import { defineConfig } from 'vite';
import react from '@vitejs/plugin-react';

// https://vitejs.dev/config/
export default defineConfig({
  base: '/lokaz/', // <-- AJOUT IMPORTANT POUR GITHUB PAGES !
  plugins: [react()],
  optimizeDeps: {
    exclude: ['lucide-react'],
  },
});

npm run build
